                                  let throttle = (fn, interval) => {
    if (typeof fn !== 'function' || typeof interval !== 'number') return false
    let prev = null
    let count = 0
    return function () {
        const arg = arguments
        const that = this
        const now = +new Date()
        count++
        if (now - prev >= interval || count === 5) {
            fn.apply(that, arg)
            prev = now
            count = 0
        }
    }
}
